# docker をインストール or docker engine を起動
https://docs.docker.com/engine/install/

# イメージ作成
Dockerfileの置かれているカレントディレクトリ (Docker) で以下を実行

docker build -t mm-enshu .


# コンテナ作成 (イメージ作成と同じカレントディレクトリで実行)
docker run -dit --name mm-enshu_test mm-enshu
(コンテナの root/code とイメージ作成時のカレントディレクトリが同期)


docker run -dit -v $PWD:/code mm-enshu_test mm-enshu

#コンテナに入る
docker container exec -it mm-enshu_test bash
